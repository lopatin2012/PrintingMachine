📚 Справка по синтаксису шаблонов

ZPL (Zebra Programming Language)

^XA                    // Начало этикетки
^FO50,50              // Позиция (X,Y)
^A0N,30,30            // Шрифт (стандартный, нормальный, высота 30, ширина 30)
^FDТекст^FS           // Данные и конец поля
^XZ                    // Конец этикетки

TSPL (TSC Programming Language)

SIZE 60 mm, 40 mm      // Размер этикетки
GAP 2 mm, 0 mm         // Зазор между этикетками
DIRECTION 0            // Направление печати
REFERENCE 0,0          // Точка отсчёта
CLS                    // Очистка буфера
TEXT 10,10,"1",0,1,1,"Текст"
PRINT 1,1              // Печать 1 копии
